#include <stdio.h>
#include <unistd.h>
#include <string.h>
#include <sys/wait.h>
#include <sys/types.h>

int n;

int fib(int n) {
	if (n == 1 || n == 2)
		return 1;
	return fib(n-1)+fib(n-2);
}
  
int main() {

	pid_t t;
	t = fork();

	if (t == 0) {
		printf("n = ");
		scanf("%d", &n);

		_exit(0);
	}

	else if (t > 0) {
		int crc;
		wait(&crc);

		for (int i = 1; i <= n; i++)
			printf("%d ", fib(i));
		printf("\n");
	}

	else {printf("Error");}

	return 0;
}
